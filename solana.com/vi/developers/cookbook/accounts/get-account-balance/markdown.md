# [How to Get Account Balance](/vi/developers/cookbook/accounts/get-account-
balance)

get-account-balance.ts

    
    
    import {
      clusterApiUrl,
      Connection,
      PublicKey,
      LAMPORTS_PER_SOL,
    } from "@solana/web3.js";
     
    (async () => {
      const connection = new Connection(clusterApiUrl("devnet"), "confirmed");
     
      let wallet = new PublicKey("G2FAbFQPFa5qKXCetoFZQEvF9BVvCKbvUZvodpVidnoY");
      console.log(
        `${(await connection.getBalance(wallet)) / LAMPORTS_PER_SOL} SOL`,
      );
    })();
